### B2005 字符三角形

```cpp
#include <iostream>
using namespace std;
int main()
{
	//freopen("in.txt", "r", stdin);
	//freopen("out.txt", "w", stdout);

	char c;
	cin >> c;
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 2 - i; j++)
		{
			cout << " ";
		}
		for (int j = 0; j < 2 * i + 1; j++)
		{
			cout << c;
		}
		cout << endl;
	}

	//fclose(stdin);
	//fclose(stdout);
	return 0;
}
```

### P5708 【深基2.习2】三角形面积

```cpp
#include <iostream>
#include <cmath>
#include <iomanip>

using namespace std;
int main()
{
	//freopen("in.txt", "r", stdin);
	//freopen("out.txt", "w", stdout);

	double a, b, c;
	cin >> a >> b >> c;
	double p = (a + b + c) / 2;
	double area = sqrt(p * (p - a) * (p - b) * (p - c));
	cout << fixed << setprecision(1) << area;
	cout << endl;
```

```cpp
#include <iomanip>
using namespace std;

{
	//freopen("in.txt", "r", stdin);
	//freopen("out.txt", "w", stdout);

	int s, v;
	cin >> s >> v;
	int time;
	time = 10 + s / v + (bool)(s % v);
	//cout << "time=" << time << endl;
	int time_hour = time / 60;
	int time_minute = time % 60;
	int ans_hour, ans_minute;
	ans_hour = 8 - time_hour;
	ans_minute = 0 - time_minute;
	ans_hour--;
	ans_minute += 60;
	if (ans_hour < 0)
	{
		ans_hour += 24;
	}
	cout << setfill('0') << setw(2) << ans_hour << ":" << setw(2) << ans_minute << endl;

	//fclose(stdin);
	//fclose(stdout);
	return 0;
}
```
