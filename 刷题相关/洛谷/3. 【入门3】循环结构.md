vector<int> times(vector<int>& a, vector<int>& b)
{
	int la = a.size();
	int lb = b.size();
	if (la < lb)//保证a是长数
	{
		swap(a, b);
		swap(la, lb);
	}
	vector<int> ans;
	for (int i = 0; i < la + lb; i++)
		ans.push_back(0);
	int temp[2];
	for (int i = lb - 1; i >= 0; i--)
	{
		for (int j = la - 1; j >= 0; j--)
		{
			temp[0] = a[j] * b[i] % 10;
			temp[1] = a[j] * b[i] / 10;
			ans[i + j + 1] += temp[0];
			ans[i + j] += temp[1];
		}
	}
	for (int i = ans.size() - 1; i > 0; i--)
	{
		ans[i - 1] += ans[i] / 10;
		ans[i] %= 10;
	}
	return ans;
}

vector<int> add(vector<int>& a, vector<int>& b)
{
	int len = max(a.size(), b.size());
	vector<int> ans;
	for (int i = 0; i < len+1; i++)//结果的位数取二者最大值+1
	{
		ans.push_back(0);
	}
	for (int i = 1; i <= len; i++)//求和
	{
		if ((int)a.size() - i >= 0)
			ans[ans.size() - i] += a[a.size() - i];
		if ((int)b.size() - i >= 0)
			ans[ans.size() - i] += b[b.size() - i];
	}
	for (int i = ans.size() - 1; i > 0; i--)//将进位计算出来
	{
		if (ans[i] > 9)
		{
			ans[i - 1] += ans[i] / 10;
			ans[i] %= 10;
		}
	}
	while (ans[0] == 0)
	{
		vector<int>::iterator it = ans.begin();
		ans.erase(it);
	}
	return ans;
}

bool isPrime(int num){
    if(num <= 1) return false;
    if(num == 2) return true;
    for(int i = 3; i <= sqrt(num); i += 2){
        if(num % i == 0)
            return false;
    }
    return true;
}
bool isPalindrome(int num){
    string num_s = to_string(num);
    while(1){
        if(num_s.length() <= 1) return true;
        if(num_s[0] == num_s[num_s.length() - 1]){
            num_s.pop_back();
            num_s.erase(0,1);
        }
        else return false;
    }
}
vector<int> getPalindrome(const int length){
    vector<int> palindromes;
    const int minNum = int(pow(10, (length + 1) / 2 - 1));
    const int maxNum = int(pow(10, (length + 1) / 2)) - 1;
    //cout << minNum << endl << maxNum << endl;
    for(int i = minNum; i <= maxNum; i++){
        string i_str = to_string(i);
        //cout << i_str << endl;
        string new_string = i_str;
        //cout << new_string << endl;
        reverse(new_string.begin(),new_string.end());
        //cout << new_string << endl;
        if(length % 2 != 0) new_string.erase(0,1);
        i_str += new_string;
        //cout << i_str << endl;
        //cout << stoi(i_str) << endl;
        palindromes.push_back(stoi(i_str));
    }
    return palindromes;
}

    const int length_max = to_string(b).length();
