### P5717 【深基3.习8】三角形分类

```cpp
#include <iostream>
using namespace std;
bool is_triangle(int, int, int);
bool is_right(int, int, int);
bool is_acute(int, int, int);
bool is_obtuse(int, int, int);
bool is_isosceles(int, int, int);
bool is_equilateral(int, int, int);
int main()
{
	//freopen("in.txt", "r", stdin);
	//freopen("out.txt", "w", stdout);

	int a, b, c;
	cin >> a >> b >> c;
	if (is_triangle(a, b, c))
	{
		if (is_right(a, b, c))
		{
			cout << "Right triangle" << endl;
		}
		else if (is_acute(a, b, c))
		{
			cout << "Acute triangle" << endl;
		}
		else
		{
			cout << "Obtuse triangle" << endl;
		}
		if (is_isosceles(a, b, c))
		{
			cout << "Isosceles triangle" << endl;
			if (is_equilateral(a, b, c))
			{
				cout << "Equilateral triangle" << endl;
			}
		}
	}
	else
	{
		cout << "Not triangle" << endl;
	}
	//fclose(stdin);
	//fclose(stdout);
	return 0;
}
bool is_triangle(int a, int b, int c)
{
	bool ans = 1;
	if (a <= 0 || b <= 0 || c <= 0) ans = 0;
	if (a + b <= c) ans = 0;
	if (b + c <= a) ans = 0;
	if (c + a <= b) ans = 0;
	return ans;
}
bool is_right(int a, int b, int c)
{
	bool ans = 0;
	if (a * a == b * b + c * c) ans = 1;
	if (b * b == a * a + c * c) ans = 1;
	if (c * c == b * b + a * a) ans = 1;
	return ans;
}
bool is_acute(int a, int b, int c)
{
	int ans = 0;
	if (a * a < b * b + c * c) ans += 1;
	if (b * b < a * a + c * c) ans += 1;
	if (c * c < b * b + a * a) ans += 1;
	if (ans == 3) return true;
	else return false;
}
bool is_obtuse(int a, int b, int c)
{
	bool ans = 0;
	if (!is_right(a, b, c) && !is_acute(a, b, c))
	{
		ans = 1;
	}
	return ans;
}
bool is_isosceles(int a, int b, int c)
{
	bool ans = 0;
	if (a == b) ans = 1;
	if (b == c) ans = 1;
	if (a == c) ans = 1;
	return ans;
}
bool is_equilateral(int a, int b, int c)
{
	bool ans = 1;
	if (a != b) ans = 0;
	if (b != c) ans = 0;
	if (a != c) ans = 0;
	return ans;
}
```
