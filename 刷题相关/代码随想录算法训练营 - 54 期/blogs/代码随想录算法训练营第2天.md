class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int left = 0, right = 0;
        int sum = nums[0];
        int ans = nums.size() + 1;
        while(right < nums.size()){
            if(sum < target){
                right ++;
                if(right == nums.size()) break;
                sum += nums[right];
            }else{
                if(ans > (right - left + 1)){
                    ans = right - left + 1;
                }
                sum -= nums[left];
                left ++;
            }
        }
        return ans == nums.size() + 1 ? 0 : ans;
    }
};
