// https://leetcode.cn/problems/minimum-size-subarray-sum/

class Solution
{
public:
    int minSubArrayLen(int target, vector<int> &nums)
    {
        // 先定义一个最小长度值
        int min_length = 1 << 31 - 1;
        // 定义窗口内的值总和
        int window_sum = nums[0];
        // 定义窗口长度
        int window_size = 1;
        // 定义双指针
        int left_target = 0, right_target = 0;
        while (right_target < nums.size() && left_target <= right_target)
        {
            // 先判断窗口内的值是否满足要求
            if (window_sum >= target)
            {
                //更新最小值
                if (window_size < min_length)
                {
                    min_length = window_size;
                }
                // 满足要求，右移左指针，更新相关变量
                window_sum -= nums[left_target];
                left_target++;
                window_size--;
            }
            else
            {
                // 不满足要求，右移右指针，更新相关变量
                right_target++;
                // 考虑边界情况
                if (right_target == nums.size())
                {
                    break;
                }
                window_sum += nums[right_target];
                window_size++;
            }
        }
        if (min_length == 1 << 31 - 1)
            return 0;
        return min_length;
    }
};
