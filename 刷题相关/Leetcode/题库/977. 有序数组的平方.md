```C++
// https://leetcode.cn/problems/squares-of-a-sorted-array/

class Solution
{
public:
    vector<int> sortedSquares(vector<int> &nums)
    {
        vector<int> ans;
        int left_index = 0, right_index = nums.size() - 1;
        while(left_index <= right_index){
            if(abs(nums[left_index]) >= abs(nums[right_index])){
                ans.push_back(nums[left_index] * nums[left_index]);
                left_index++;
            }else{
                ans.push_back(nums[right_index] * nums[right_index]);
                right_index--;
            }
        }
        //翻转ans
        int temp;
        for(int i = 0; i < ans.size()/2; i++){
            temp = ans[i];
            ans[i] = ans[ans.size() - i - 1];
            ans[ans.size() - i - 1] = temp;
        }
        return ans;
    }
};
```
