class Solution {
public:
    int mySqrt(int x) {
        //边界条件确定
        if(x == 0) return 0;
        if(x == 1) return 1;
        //使用二分查找的思想，对结果进行筛选
        //首先确定边界
        int left = 0, right = x;
        int middle;
        while(left <= right){
            middle = (left + right) / 2;
            if(double(x/middle) == middle){
                return middle;
            }
            else if(double(x/middle) > middle){
                left = middle + 1;
            }
            else{
                right = middle - 1;
            }
        }
        if(middle * middle > x){
            return middle - 1;
        }
        else{
            return middle;
        }
    }
};
