## 代码实现

```Python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        # 先处理特殊情况(链表长度不足3)
        if head is None:
            return head
        elif head.next is None:
            return head
        elif head.next.next is None:
            temp = head.next
            temp.next = head
            head.next = None
            return temp
        # 经过分析，设置 3 个指针来进行操作是比较简单直观的
        pre = head
        cur = pre.next
        nxt = cur.next
        pre.next = None
        # 设置循环条件
        while nxt is not None:
            cur.next = pre
            pre = cur
            cur = nxt
            nxt = nxt.next
        cur.next = pre
        return cur
```
